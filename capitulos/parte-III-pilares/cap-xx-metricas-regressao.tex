\section{Raíz do Erro Quadrático Médio (RMSE)}
\label{sec:rmse}

\begin{equacaodestaque}{Raíz do Erro Quadrático Médio (RMSE)}
    RMSE = \sqrt{\frac{1}{n} \sum_{i=1}^{n} (y_i - \hat{y}_i)^2}
    \label{eq:equacao-rmse}
\end{equacaodestaque}

\section{Erro Percentual Absoluto Médio (MAPE)}
\label{sec:mape}

\begin{equacaodestaque}{Erro Percentual Absoluto Médio (MAPE)}
    MAPE = \frac{1}{n} \sum_{i=1}^{n} \left| \frac{y_i - \hat{y}_i}{y_i} \right|
    \label{eq:equacao-mape}
\end{equacaodestaque}

\section{Erro Percentual Absoluto Médio Simétrico (SMAPE)}
\label{sec:smape}

\begin{equacaodestaque}{Erro Percentual Absoluto Médio Simétrico (SMAPE)}
    SMAPE = \frac{1}{n} \sum_{i=1}^{n} \frac{2 |y_i - \hat{y}_i|}{|y_i| + |\hat{y}_i|}
    \label{eq:equacao-smape}
\end{equacaodestaque}

\section{Coeficiente de Determinação R2}
\label{sec:r2}

\begin{equacaodestaque}{Coeficiente de Determinação ($R^2$)}
    R^2 = 1 - \frac{\sum_{i=1}^{n} (y_i - \hat{y}_i)^2}{\sum_{i=1}^{n} (y_i - \bar{y})^2}
    \label{eq:equacao-r2}
\end{equacaodestaque}
