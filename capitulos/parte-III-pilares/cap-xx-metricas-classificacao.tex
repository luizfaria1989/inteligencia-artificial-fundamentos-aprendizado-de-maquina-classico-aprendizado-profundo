\section{Precisão (Precision), Valor Preditivo Positivo (PPV)}
\label{sec:precisao}
% O usuário listou TDR, que é True Discovery Rate, mas é menos comum que PPV.
% Mantive PPV que é sinônimo de Precisão.

\begin{equacaodestaque}{Precisão (\textit{Precision})}
    Precisão = \frac{VP}{VP + FP}
    \label{eq:equacao-precisao}
\end{equacaodestaque}

\section{F1-Score}
\label{sec:f1-score}

\begin{equacaodestaque}{F1-Score}
    F_1 = 2 \cdot \frac{\text{Precisão} \cdot \text{Recall}}{\text{Precisão} + \text{Recall}}
    \label{eq:equacao-f1-score}
\end{equacaodestaque}
